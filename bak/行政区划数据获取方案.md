## 二、阿里云DataV API详解

### 2.1 API格式

```
基础URL: https://geo.datav.aliyun.com/areas_v3/bound/

接口格式:
- 行政区列表: {adcode}.json
- 完整边界: {adcode}_full.json
- 简化边界: {adcode}_simplified.json
```

### 2.2 行政区划代码(adcode)规则

| 级别 | 编码规则 | 示例 |
|-----|---------|------|
| 国家 | 100000 | 100000 |
| 省级 | XX0000 | 110000(北京)、310000(上海) |
| 地级市 | XXXX00 | 110100(北京市)、320100(南京) |
| 区县级 | XXXXXX | 110101(东城区)、320102(玄武区) |

## 三、数据获取脚本

### 3.1 批量下载全国行政区GeoJSON

```javascript
// download_geojson.js
const fs = require('fs');
const https = require('https');
const path = require('path');

const BASE_URL = 'https://geo.datav.aliyun.com/areas_v3/bound/';
const OUTPUT_DIR = './geojson_data';

// 创建输出目录
if (!fs.existsSync(OUTPUT_DIR)) {
    fs.mkdirSync(OUTPUT_DIR, { recursive: true });
}

// 下载函数
function downloadGeoJSON(adcode, withChildren = true) {
    return new Promise((resolve, reject) => {
        const suffix = withChildren ? '_full.json' : '.json';
        const url = `${BASE_URL}${adcode}${suffix}`;
        const filename = `${adcode}${suffix}`;
        const filepath = path.join(OUTPUT_DIR, filename);
        
        console.log(`Downloading: ${url}`);
        
        https.get(url, (res) => {
            if (res.statusCode === 404) {
                console.log(`Not found: ${adcode}`);
                resolve(null);
                return;
            }
            
            let data = '';
            res.on('data', chunk => data += chunk);
            res.on('end', () => {
                try {
                    fs.writeFileSync(filepath, data, 'utf8');
                    console.log(`Saved: ${filename}`);
                    resolve(JSON.parse(data));
                } catch (e) {
                    reject(e);
                }
            });
        }).on('error', reject);
    });
}

// 获取所有省级行政区
async function getProvinces() {
    const data = await downloadGeoJSON(100000, false);
    if (data && data.features) {
        return data.features.map(f => ({
            adcode: f.properties.adcode,
            name: f.properties.name
        }));
    }
    return [];
}

// 递归下载所有层级
async function downloadAll() {
    // 1. 获取省列表
    const provinces = await getProvinces();
    console.log(`Found ${provinces.length} provinces`);
    
    // 2. 下载每个省的完整数据
    for (const province of provinces) {
        await downloadGeoJSON(province.adcode, true);
        await sleep(100); // 避免请求过快
    }
}

function sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
}

downloadAll().catch(console.error);
```

### 3.2 Python版本下载脚本

```python
# download_geojson.py
import requests
import json
import os
import time

BASE_URL = 'https://geo.datav.aliyun.com/areas_v3/bound/'
OUTPUT_DIR = './geojson_data'

os.makedirs(OUTPUT_DIR, exist_ok=True)

def download_geojson(adcode, with_children=True):
    suffix = '_full.json' if with_children else '.json'
    url = f'{BASE_URL}{adcode}{suffix}'
    filepath = os.path.join(OUTPUT_DIR, f'{adcode}{suffix}')
    
    try:
        print(f'Downloading: {url}')
        response = requests.get(url, timeout=30)
        
        if response.status_code == 404:
            print(f'Not found: {adcode}')
            return None
        
        data = response.json()
        with open(filepath, 'w', encoding='utf-8') as f:
            json.dump(data, f, ensure_ascii=False, indent=2)
        
        print(f'Saved: {filepath}')
        return data
    except Exception as e:
        print(f'Error downloading {adcode}: {e}')
        return None

def get_all_provinces():
    data = download_geojson(100000, with_children=False)
    if data and 'features' in data:
        return [(f['properties']['adcode'], f['properties']['name']) 
                for f in data['features']]
    return []

def download_all():
    provinces = get_all_provinces()
    print(f'Found {len(provinces)} provinces')
    
    for adcode, name in provinces:
        print(f'Processing: {name} ({adcode})')
        download_geojson(adcode, with_children=True)
        time.sleep(0.1)

if __name__ == '__main__':
    download_all()
```

### 5.2 点在多边形内判断算法

```javascript
/**
 * 判断点是否在多边形内 (射线法)
 * @param {Array} point - 点坐标 [lng, lat]
 * @param {Array} polygon - 多边形坐标数组 [[[lng, lat], ...]]
 * @returns {Boolean}
 */
function isPointInPolygon(point, polygon) {
    const [lng, lat] = point;
    let inside = false;
    
    for (let i = 0, j = polygon.length - 1; i < polygon.length; j = i++) {
        const [lngi, lati] = polygon[i];
        const [lngj, latj] = polygon[j];
        
        if (((lati > lat) !== (latj > lat)) &&
            (lng < (lngj - lngi) * (lat - lati) / (latj - lati) + lngi)) {
            inside = !inside;
        }
    }
    
    return inside;
}

/**
 * 判断点是否在MultiPolygon内
 */
function isPointInMultiPolygon(point, geometry) {
    if (geometry.type === 'Polygon') {
        return isPointInPolygon(point, geometry.coordinates[0]);
    } else if (geometry.type === 'MultiPolygon') {
        return geometry.coordinates.some(polygon => 
            isPointInPolygon(point, polygon[0])
        );
    }
    return false;
}
```

### 5.3 邮编定位服务实现

```javascript
// postcode_service.js
class PostcodeLocator {
    constructor(geojsonDir) {
        this.geojsonDir = geojsonDir;
        this.geojsonCache = new Map();
    }
    
    // 加载GeoJSON数据
    async loadGeoJSON(adcode) {
        if (this.geojsonCache.has(adcode)) {
            return this.geojsonCache.get(adcode);
        }
        
        const fs = require('fs');
        const path = require('path');
        const filepath = path.join(this.geojsonDir, `${adcode}_full.json`);
        
        if (fs.existsSync(filepath)) {
            const data = JSON.parse(fs.readFileSync(filepath, 'utf8'));
            this.geojsonCache.set(adcode, data);
            return data;
        }
        return null;
    }
    
    // 根据邮编定位
    async locate(postcode) {
        // 1. 查询邮编位置数据
        const postcodeData = await this.getPostcodeData(postcode);
        if (!postcodeData) {
            return { success: false, message: '邮编不存在' };
        }
        
        const { longitude, latitude, adcode, province, city, district } = postcodeData;
        const point = [longitude, latitude];
        
        // 2. 加载对应行政区GeoJSON
        const geojson = await this.loadGeoJSON(adcode);
        if (!geojson) {
            return { success: false, message: '边界数据不存在' };
        }
        
        // 3. 查找点所在的区域
        for (const feature of geojson.features) {
            if (isPointInMultiPolygon(point, feature.geometry)) {
                return {
                    success: true,
                    data: {
                        postcode,
                        location: { longitude, latitude },
                        adminArea: {
                            adcode: feature.properties.adcode,
                            name: feature.properties.name,
                            level: feature.properties.level,
                            center: feature.properties.center
                        },
                        boundary: feature.geometry
                    }
                };
            }
        }
        
        // 4. 如果没找到精确匹配，返回区县级信息
        return {
            success: true,
            data: {
                postcode,
                location: { longitude, latitude },
                adminArea: {
                    adcode,
                    name: district,
                    level: 'district',
                    province,
                    city
                }
            }
        };
    }
    
    // 获取邮编数据(模拟)
    async getPostcodeData(postcode) {
        // 实际应从数据库查询
        const postcodeMap = {
            '100000': {
                postcode: '100000',
                province: '北京市',
                city: '北京市',
                district: '东城区',
                longitude: 116.418757,
                latitude: 39.917544,
                adcode: '110101'
            }
        };
        return postcodeMap[postcode];
    }
}

module.exports = PostcodeLocator;
```

### 9.2 坐标系说明

- 阿里云DataV数据使用 **GCJ-02坐标系** (火星坐标系)
- GPS原始坐标为 **WGS-84坐标系**
- 百度地图使用 **BD-09坐标系**

需要进行坐标转换：

```javascript
// GCJ-02 转 WGS-84
function gcj02ToWgs84(lng, lat) {
    const PI = 3.14159265358979324;
    const a = 6378245.0;
    const ee = 0.00669342162296594323;
    
    let dLat = transformLat(lng - 105.0, lat - 35.0);
    let dLng = transformLng(lng - 105.0, lat - 35.0);
    
    const radLat = lat / 180.0 * PI;
    let magic = Math.sin(radLat);
    magic = 1 - ee * magic * magic;
    const sqrtMagic = Math.sqrt(magic);
    
    dLat = (dLat * 180.0) / ((a * (1 - ee)) / (magic * sqrtMagic) * PI);
    dLng = (dLng * 180.0) / (a / sqrtMagic * Math.cos(radLat) * PI);
    
    return {
        longitude: lng * 2 - (lng + dLng),
        latitude: lat * 2 - (lat + dLat)
    };
}
```

### 9.3 数据更新频率

- 邮编数据: 相对稳定，变化较少
